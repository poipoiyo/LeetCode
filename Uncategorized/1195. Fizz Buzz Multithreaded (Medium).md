## **1195. Fizz Buzz Multithreaded (Medium)** 

https://leetcode.com/problems/fizz-buzz-multithreaded/



#### Ideas : 

1. Solution 1 takes four `mutex` to control threads.  We count `i` again and unlock correspond mutex after every print.
2. Solution 2 takes `condition_variable` to replace multi mutex. 



#### Solution 1 :

Language : C++

```C++
class FizzBuzz 
{
private:
    int n, i;
    mutex mF, mB, mFB, mN;
    
    void count()
    {
        ++i;
        if (i > n) 
        {
            mF.unlock();
            mB.unlock();
            mFB.unlock();
            mN.unlock();
        }
        else if (i % 15 == 0)
            mFB.unlock();
        else if (i % 3 == 0)
            mF.unlock();
        else if (i % 5 == 0)
            mB.unlock();
        else 
            mN.unlock();
        
    }

public:
    FizzBuzz(int n) 
    {
        this->n = n;
        
        i = 1;
        mF.lock();
        mB.lock();
        mFB.lock();
    }

    // printFizz() outputs "fizz".
    void fizz(function<void()> printFizz) 
    {
       while(i <= n) 
       {
            mF.lock();
            if (i <= n) 
            {
                printFizz();
                count();
            }
        }
    }

    // printBuzz() outputs "buzz".
    void buzz(function<void()> printBuzz) 
    {
        while(i <= n) 
        {
            mB.lock();
            if (i <= n) 
            {
                printBuzz();
                count();
            }
        }
    }

    // printFizzBuzz() outputs "fizzbuzz".
	void fizzbuzz(function<void()> printFizzBuzz) 
    {
        while(i <= n) 
        {
            mFB.lock();
            if (i <= n) 
            {
                printFizzBuzz();
                count();
            }
        }
    }

    // printNumber(x) outputs "x", where x is an integer.
    void number(function<void(int)> printNumber) 
    {
        while(i <= n) 
        {
            mN.lock();
            if (i <= n) 
            {
                printNumber(i);
                count();
            }
        }
    }
};
```



#### Solution 2 :

Language : C++

```c++
class FizzBuzz 
{
private:
    int n;
    int count;
    mutex m;
    condition_variable cv;

public:
    FizzBuzz(int n) 
    {
        this->n = n;
        this->count = 1;
    }

    // printFizz() outputs "fizz".
    void fizz(function<void()> printFizz) 
    {
        while (true) 
        {
            unique_lock<mutex> lock(m);
            cv.wait(lock, [&](){ 
                return (count>n || (count%3 == 0) && (count%5 != 0) );
            });
            
            if (count <= n) 
            {
                printFizz();
                count++;
            } 
            else 
                break; 
            cv.notify_all();
        }
    }

    // printBuzz() outputs "buzz".
    void buzz(function<void()> printBuzz) 
    {
        while (true) 
        {
            unique_lock<mutex> lock(m);
            cv.wait(lock, [&](){ 
                return (count>n || (count%3 != 0) && (count%5 == 0) );
            });
            
            if (count <= n) 
            {
                printBuzz();
                count++;
            } 
            else 
                break; 
            cv.notify_all();
        }
    }

    // printFizzBuzz() outputs "fizzbuzz".
	void fizzbuzz(function<void()> printFizzBuzz) 
    {
        while (true) 
        {
            unique_lock<mutex> lock(m);
            cv.wait(lock, [&](){ 
                return (count>n || (count%3 == 0) && (count%5 == 0) );
            });
            
            if (count <= n) 
            {
                printFizzBuzz();
                count++;
            } 
            else 
                break; 
            cv.notify_all();
        }
    }

    // printNumber(x) outputs "x", where x is an integer.
    void number(function<void(int)> printNumber) 
    {
        while (true) 
        {
            unique_lock<mutex> lock(m);
            cv.wait(lock, [&](){ 
                return (count>n || (count%3 != 0) && (count%5 != 0) );
            });
            
            if (count <= n) 
            {
                printNumber(count);
                count++;
            } 
            else 
                break; 
            cv.notify_all();
        }
    }
};
```
